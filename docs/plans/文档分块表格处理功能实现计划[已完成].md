# 文档分块表格处理功能实现计划[已完成]

## Context（背景）

当前 `document_chunking.py` 只处理 Word 文档的段落内容（`document.paragraphs`），完全忽略表格。这导致：

1. **表格内容丢失**：表格中的所有数据无法被检索到
2. **信息完整性缺失**：金融保险场景中，表格通常包含关键的条款、费率、保障范围等结构化信息

用户提出的解决方案：
- 将表格转换为 Markdown 格式，便于检索和展示
- 处理合并单元格（跳过 `w:vMerge` 标记的延续格）
- 表格独立分块，超长表格按行分组并补充表头

## 实现方案

### 1. 数据结构调整

在文件顶部添加类型定义，用于区分段落和表格：

```python
from enum import Enum
from dataclasses import dataclass

class SegmentType(Enum):
    """文档片段类型。"""
    PARAGRAPH = "paragraph"
    TABLE = "table"

@dataclass
class DocumentSegment:
    """文档片段，可以是段落或表格。"""
    content: str
    segment_type: SegmentType
    heading_context: str
```

### 2. 核心改动：重构 `_extract_docx_segments()`

**当前问题**：只遍历 `document.paragraphs`，无法获取表格。

**解决方案**：遍历 `document.element.body` 的所有子元素，按原始顺序处理段落和表格。

```python
from docx.oxml.table import CT_Tbl
from docx.table import Table

def _extract_docx_segments(self, document: DocxDocument) -> list[DocumentSegment]:
    """抽取段落和表格，保持原始顺序。"""
    headings: list[str] = []
    segments: list[DocumentSegment] = []

    for element in document.element.body:
        if isinstance(element, CT_P):
            # 处理段落（复用现有逻辑）
            paragraph = Paragraph(element, document)
            segment = self._process_paragraph(paragraph, headings)
            if segment:
                segments.append(segment)
        elif isinstance(element, CT_Tbl):
            # 处理表格（新增逻辑）
            table = Table(element, document)
            segment = self._process_table(table, headings)
            if segment:
                segments.append(segment)

    return segments
```

### 3. 新增方法：段落处理

将原 `_extract_docx_segments()` 中的段落处理逻辑提取为独立方法：

```python
def _process_paragraph(
    self,
    paragraph: Paragraph,
    headings: list[str]
) -> DocumentSegment | None:
    """处理单个段落，返回 DocumentSegment。"""
    text = paragraph.text.strip()
    if not text:
        return None

    heading_level = self._get_heading_level(paragraph)
    if heading_level is not None:
        headings[:] = headings[: heading_level - 1]
        headings.append(text)
        content = " / ".join(headings)
    else:
        context = " / ".join(headings)
        content = f"{context}\n{text}" if context else text

    return DocumentSegment(
        content=content,
        segment_type=SegmentType.PARAGRAPH,
        heading_context=" / ".join(headings),
    )
```

### 4. 新增方法：表格处理

```python
def _process_table(
    self,
    table: Table,
    headings: list[str]
) -> DocumentSegment | None:
    """处理单个表格，转换为 Markdown 格式。"""
    markdown = self._table_to_markdown(table)
    if not markdown:
        return None

    return DocumentSegment(
        content=markdown,
        segment_type=SegmentType.TABLE,
        heading_context=" / ".join(headings),
    )
```

### 5. 新增方法：表格转 Markdown

```python
def _table_to_markdown(self, table: Table) -> str:
    """将表格转换为 Markdown 格式，处理合并单元格。"""
    if not table.rows:
        return ""

    lines = []

    # 处理表头（第一行）
    header_cells = []
    for cell in table.rows[0].cells:
        if not self._is_vertically_merged_continuation(cell):
            header_cells.append(cell.text.strip().replace("\n", " "))
        else:
            header_cells.append("")

    lines.append("| " + " | ".join(header_cells) + " |")
    lines.append("|" + "|".join(["---"] * len(header_cells)) + "|")

    # 处理数据行
    for row in table.rows[1:]:
        cells = []
        for cell in row.cells:
            if not self._is_vertically_merged_continuation(cell):
                cells.append(cell.text.strip().replace("\n", " "))
            else:
                cells.append("")
        lines.append("| " + " | ".join(cells) + " |")

    return "\n".join(lines)

def _is_vertically_merged_continuation(self, cell) -> bool:
    """检查单元格是否是垂直合并的延续格（需要跳过）。"""
    tc_pr = cell._element.tcPr
    if tc_pr is None:
        return False

    vmerge = tc_pr.vMerge
    if vmerge is None:
        return False

    # restart 表示合并开始，continue 或无 val 表示延续
    val = vmerge.val
    return val is None or val == "continue"
```

### 6. 重构 `chunk_docx()` 方法

**当前逻辑**：所有 segments 拼接成字符串，滑动窗口切块。

**新逻辑**：段落和表格分别处理。

```python
def chunk_docx(
    self,
    file_path: Path,
    source_filename: str,
    storage_key: str,
    file_id: str,
) -> list[DocumentChunk]:
    try:
        document = Document(str(file_path))
    except Exception as exc:
        msg = f"解析 docx 文件失败: {source_filename}"
        raise DocumentParseError(msg) from exc

    segments = self._extract_docx_segments(document)
    if not segments:
        msg = f"Word 文档内容为空，无法切块: {source_filename}"
        raise DocumentParseError(msg)

    chunks: list[DocumentChunk] = []
    paragraph_buffer: list[str] = []

    for segment in segments:
        if segment.segment_type == SegmentType.PARAGRAPH:
            paragraph_buffer.append(segment.content)
        elif segment.segment_type == SegmentType.TABLE:
            # 先处理累积的段落
            if paragraph_buffer:
                para_text = "\n\n".join(paragraph_buffer)
                para_chunks = self._build_chunks(
                    text=para_text,
                    source_filename=source_filename,
                    storage_key=storage_key,
                    file_id=file_id,
                )
                chunks.extend(para_chunks)
                paragraph_buffer.clear()

            # 处理表格
            table_chunks = self._build_table_chunks(
                table_markdown=segment.content,
                source_filename=source_filename,
                storage_key=storage_key,
                file_id=file_id,
            )
            chunks.extend(table_chunks)

    # 处理剩余段落
    if paragraph_buffer:
        para_text = "\n\n".join(paragraph_buffer)
        para_chunks = self._build_chunks(
            text=para_text,
            source_filename=source_filename,
            storage_key=storage_key,
            file_id=file_id,
        )
        chunks.extend(para_chunks)

    # 重新编号
    for idx, chunk in enumerate(chunks):
        object.__setattr__(chunk, 'chunk_index', idx)
        object.__setattr__(chunk, 'chunk_id', f"{file_id}-chunk-{idx}")

    return chunks
```

### 7. 新增方法：表格分块

```python
def _build_table_chunks(
    self,
    table_markdown: str,
    source_filename: str,
    storage_key: str,
    file_id: str,
) -> list[DocumentChunk]:
    """为表格生成独立的 chunks。

    小表格整体作为一个 chunk；
    大表格按行分组切分，每组补充表头。
    """
    if len(table_markdown) <= self._chunk_size:
        # 小表格，整体作为一个 chunk
        term_match_result = self._term_matcher.extract_terms(table_markdown)
        return [
            DocumentChunk(
                file_id=file_id,
                chunk_id="",
                chunk_index=0,
                content=table_markdown,
                char_count=len(table_markdown),
                source_filename=source_filename,
                storage_key=storage_key,
                merged_terms=term_match_result.merged_terms,
            )
        ]

    # 大表格，需要切分
    lines = table_markdown.split("\n")
    if len(lines) < 3:
        # 格式异常，当作小表格
        term_match_result = self._term_matcher.extract_terms(table_markdown)
        return [
            DocumentChunk(
                file_id=file_id,
                chunk_id="",
                chunk_index=0,
                content=table_markdown,
                char_count=len(table_markdown),
                source_filename=source_filename,
                storage_key=storage_key,
                merged_terms=term_match_result.merged_terms,
            )
        ]

    header_line = lines[0]
    separator_line = lines[1]
    split_tables = self._split_large_table(table_markdown, header_line, separator_line)

    chunks = []
    for table_part in split_tables:
        term_match_result = self._term_matcher.extract_terms(table_part)
        chunks.append(
            DocumentChunk(
                file_id=file_id,
                chunk_id="",
                chunk_index=0,
                content=table_part,
                char_count=len(table_part),
                source_filename=source_filename,
                storage_key=storage_key,
                merged_terms=term_match_result.merged_terms,
            )
        )

    return chunks

def _split_large_table(
    self,
    markdown: str,
    header_line: str,
    separator_line: str
) -> list[str]:
    """将超大表格按行分组切分，每组补充表头。"""
    lines = markdown.split("\n")
    if len(lines) <= 2:
        return [markdown]

    header = f"{header_line}\n{separator_line}"
    header_size = len(header)
    data_rows = lines[2:]

    chunks = []
    current_rows = []
    current_size = header_size

    for row in data_rows:
        row_size = len(row) + 1  # +1 for newline

        if current_size + row_size > self._chunk_size and current_rows:
            chunk_content = header + "\n" + "\n".join(current_rows)
            chunks.append(chunk_content)
            current_rows = []
            current_size = header_size

        current_rows.append(row)
        current_size += row_size

    if current_rows:
        chunk_content = header + "\n" + "\n".join(current_rows)
        chunks.append(chunk_content)

    return chunks if chunks else [markdown]
```
